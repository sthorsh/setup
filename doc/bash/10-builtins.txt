  alias [navn[=verdi]]

  bg %job_id
  bind -pvX | keyseq:readline-fn | readline-cmd | -x keyseq:shell-cmd
  builtin builtin arg

  cd arg
  command command arg

  declare -pF | -f function | ±ainrx... navn=verdi
  disown job_id | pid

  echo -n arg
  enable -n builtin
  eval arg
  exec cmd arg
  exit [N]
  export [navn[=verdi]]

  fg %job_id

  getopts optstring navn arg

  hash [navn]
  help [builtin]
  history

  jobs -l

  kill -l | sig %job_id | sig pid

  let arg ...
  local -aAinrx... navn=verdi
  logout [N]

  mapfile array

  printf format arg
  pwd

  read navn ...
  readonly [navn=verdi]
  return [N]

  shift [N]
