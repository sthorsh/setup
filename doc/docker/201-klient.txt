Dockerfile
  FROM          set base image
  LABEL         add metadata to image
  ARG           define variable
  ENV           define variable persisted in image
  SHELL         set shell for RUN etc
  USER          set user for RUN etc
  WORKDIR       set working directory for RUN etc
  RUN cmd       execute shell cmd && commit layer
  ADD           add files inside context to image filesystem
  COPY          add files inside context to image filesystem
  VOLUME arg    create mount point at arg for volume
  ONBUILD       add on-build trigger to image
  EXPOSE        document port intended to be published
  CMD           execute cmd on launch
  ENTRYPOINT    execute cmd on launch
  HEALTHCHECK   check container health
  STOPSIGNAL    set signal to stop container
