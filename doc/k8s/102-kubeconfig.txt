KUBECONFIG

apiVersion: v1
clusters:
  - cluster:
      certificate-authority-data: [ca.crt]
      server: https://192.168.56.101:6443
    name: kubernetes
contexts:
  - context:
      cluster: kubernetes
      namespace: [namespace]
      user: kubernetes-admin
    name: kubernetes-admin@kubernetes
current-context: kubernetes-admin@kubernetes
users:
  - name: kubernetes-admin
    user:
      client-certificate-data: [client.crt] or [token] or [password]
      client-key-data: [private_key]

Lese ca.crt
  $ kubectl config view --raw -o jsonpath='{ .clusters[*].cluster.certificate-authority-data  }' | base64 --decode > ca.crt
  $ openssl x509 -in ca.crt -text -noout

Lese client crt
  $ kubectl config view --raw -o jsonpath='{ .users[*].user.client-certificate-data  }' | base64 --decode > client.crt
  $ openssl x509 -in client.crt -text -noout
