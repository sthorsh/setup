man

  $ man dockerfile

cmd
  info        system info
  version     version info
  events      real-time events from dockerd

  build       nytt image vha Dockerfile
  load        nytt image fra saved image
  import      nytt image fra eksportert container

  search      ls docker_hub images
  login       lagre i ~/.docker/config.json
  logout      slett i ~/.docker/config.json
  pull        nytt image fra docker_hub eller ...

  images      ls images
  history     ls image history
  inspect     ls image details
  tag         tag image
  create      lag container
  run         pull; extract; create; start; run
  save        save image som tar
  push        push image til docker_hub eller ...
  rmi         remove image(s)

  ps          ls containers
  top         ls container processes
  port        ls container port mappings
  diff        ls container fs changes
  logs        ls container logs
  stats       ls container resource usage
  inspect     ls container details
  start       start container(s); run cmd(s)
  attach      attach to container
  cp          fra/til path til/fra container:path
  exec        exec cmd in container
  export      export container 
  rename      rename container
  update      update container config/resources
  wait        wait for container(s) to stop
  pause       pause container(s)
  unpause     unpause container(s)
  restart     restart container(s)
  stop        stop container(s)
  kill        kill container(s)
  rm          remove container(s)
