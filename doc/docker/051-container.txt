CONTAINER

$ docker container
  create -it <image>            

  run <image> [cmd]             search; pull; create; start
  run -d                        detached
  run -it                       interaktiv terminal
  run -p p₀:p₁                  host_port : container_port 
  run -P                        random_host_port : container_port
  run -v d₀:d₁                  host_dir : container_dir
  run --mount src=v,target=d    host_volume, container_dir
  run --rm                      remove

  start <cnt>                   start/spinn_opp container; run CMD [el cmd]
  restart <cnt>                 -
  pause <cnt>                   -
  unpause <cnt>                 -
  stop <cnt>                    -
  kill <cnt>                    -
  rm <cnt>                      remove
  prune                         remove stopped containers
  
  attach <cnt>                  -
  commit <cnt>                  n/a
  cp <src> <dest>               host_path/container:path → container:path/host_path
  exec [-it] <cnt> <cmd>        exec cmd i [el exec inn i] container
  export <cnt> -o cnt.tar       -
  rename <cnt> name             -
  update <opts> <cnt>           cf container stats
  wait <cnt>                    -

  ls -a                         list containers
  stats                         list containers statistics
  diff <cnt>                    diff cow_fs ro_fs
  inspect <cnt>                 json
  logs [-f] <cnt>               pid 1 stdout/stderr
  port <cnt>                    port mappings
  top <cnt>                     table of processes on docker host
