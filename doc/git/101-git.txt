GIT

git add PATHSPEC                      stage untracked, modified, deleted files
git add --patch PATHSPEC              stage hunks

git blame FILE                        annotate file

git branch -avv                       list local, upstream, remote-tracking
git branch BRANCH [REF|COMMIT]        create local
git branch -t BRANCH REMOTE/REPO      create local, upstream
git branch -u REMOTE/REPO             set upstream
git branch --unset-upstream BRANCH    if remote branch is deleted
git branch -m BRANCH NEW_NAME         rename
git branch -d BRANCH                  delete local

git checkout -b b1                    create-checkout branch
git checkout BRANCH                   checkout target branch
git checkout COMMIT                   checkout commit (detached HEAD)
git checkout --track REPO/REF         start tracking remote branch                                    
git checkout HEAD FILE                wtree<index<repo
git checkout --patch HEAD             wtree<index<repo (hunks)

-------------------

git cherry-pick COMMIT
