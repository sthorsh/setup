DOCKER

CLI
  $ docker<CR>
  $ docker cmd --help
  $ docker mgmt_cmd<CR>
  $ docker mgmt_cmd cmd --help
  $ man docker
  $ man docker-cmd
  $ man dockerfile

cmds
  search      search docker hub
  run         pull && extract image; create && start container; run cmd
  ...
  build       build image
  tag         tag an image
    images      list images

    history     show image history
    rmi         remove image(s)
    save        save image as tar
    load        load image from tar
    export      export container:fs → fs.tar 
    import      create image from tar
...
    create      create container
    rename      rename container
...
    cp          cp fs ↔ container:fs
    diff        diff fs container:fs
    attach      attach to running container
    exec        exec cmd in running container
    pause       pause container
    unpause     unpause container
    restart     restart running container
    kill        kill running containers
    stop        stop running containers
    start       start stopped containers
    ps          list containers
    rm          remove containers
    logs        fetch logs of container
    port        list port mappings of container
    top         list containers running processes
    update      update container(s) config
...
    login       ref ~/.docker/config.json
    logout      ref ~/.docker/config.json
    pull        pull image|repo ← registry
    push        push image|repo → registry
...
    events      real-time events from dockerd
    wait        until one or more containers stop
...
    info        system-wide info
    inspect     low-level info on object
    stats       container(s) resource usage statistics
    version     Show the Docker version information
