Dockerfile
  FROM name:tag               base image
  RUN cmd                     /bin/sh -c cmd

  ARG name[=value]            build-time
  ENV key value               build-/runtime

  USER user[:group]           for RUN, CMD, ENTRYPOINT, runtime
  WORKDIR path                for RUN, ADD, COPY, CMD, ENTRYPOINT, runtime

  ADD from to                 host path|tar|url → container path
  COPY from to                host path → container path

  EXPOSE port                 map: cf run -p 
  VOLUME directory            mount: cf run -v

----------------------
  CMD                         execute cmd on launch
  ENTRYPOINT                  execute cmd on launch
