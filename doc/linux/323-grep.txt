GREP

Syntax
  grep opts regex [files]
  grep opts --regexp=regex ... [files]
  grep opts --file=filename ... [files]

Regex
  --basic-regexp
  --extended-regexp
  --fixed-strings
  --perl-regexp

Regex control
  --ignore-case  (-i)
  --word-regexp  (-w)
  --line-regexp  (-x)
  --invert-match (-v)

Files
  --recursive (-r)
  --dereference-recursive (-R)

Binary files
  --binary
  --binary-files=binary|text|without-match
  --devices=read|skip
  --null-data

File globbing
  --include=glob
  --exclude=glob
  --exclude-dir=glob
  --exclude-from=file

Output
  --color
  --count 
  --max-count=N
  --files-with-matches
  --files-without-match
  --silent
  --no-messages
  --line-buffered

Output prefix
  --with-filename
  --no-filename
  --label=label
  --null
  --line-number
  --byte-offset

Output line
  --initial-tab
  --only-matching
  --context=N
  --before-context=N
  --after-context=N
