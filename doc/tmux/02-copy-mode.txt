search
  /         search-forward-inc
  ?         search-backward-inc
  n         search-again
  N         search-reverse

scroll
  C-e       scroll-down
  C-d       halfpage-down
  C-f       page-down
  C-y       scroll-up
  C-b       page-up
  C-u       halfpage-up

jump
  f         jump-forward
  F         jump-backward
  t         jump-to-forward <to>
  T         jump-to-backward <to>
  ;         jump-again
  ,         jump-reverse

move
  b         previous-word
  B         previous-space
  e         next-word-end
  E         next-space-end
  g         history-top
  G         history-bottom
  h         cursor-left
  H         top-line
  j         cursor-down
  k         cursor-up
  l         cursor-right
  L         bottom-line
  M         middle-line
  w         next-word
  W         next-space
  0         start-of-line
  $         end-of-line
  ^         back-to-indentation
  {         previous-paragraph
  }         next-paragraph
  :         goto-line <line>

select
  v         visual
  V         visual line
  C-v       visual block
  o         other-end
  Space     clear-selection (vim: escape)

copy
  y         selection → tmux, clipit
  C-j       selection → tmux
  A         selection → tmux append
  Y         EOL → tmux
