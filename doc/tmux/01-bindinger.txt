tmux

  ?             list-keys

  C-a           prefix

  D             choose-client
  d             detach-client
  C-i           switch-client -p
  C-o           switch-client -n

  s             choose-tree

  c             new-window
  ,             rename-window
  C-hl          select-window (alt 1..9)
  f             find-window
  C-S-←→        swap-window
  .             move-window
  |-            split-window
  C-w           kill-window

  hjkl          select-pane
  q             display-panes (evt 1..9)
  z             resize-pane
  C-A-←↑↓→      resize-pane
  {}            swap-pane
  !             break-pane
  L             pipe-pane
  C-A-s         synchronize-panes
  x             kill-pane
  Ctrl-A-a      kill-pane(s)

  C-k           clear-history

  C-c           copy-mode-vi
    /           search ↓
    ?           search ↑
    v|V|C-v     select text
    ␣           clear selection
    y           yank text → clipit, tmux
    C-j         yank text → tmux
    A           yank text → tmux append
    Y           yank to EOL → tmux

  =             choose tmux buffer
  p             paste from tmux buffer
  C-v           paste from clipit buffer

  :             command-mode

  r             reload tmux.conf

  C-I           tpm: install
  C-U           tpm: update

  C-s           tmux-resurrect: save
  C-r           tmux-resurrect: restore
