tmux

  ?             list-keys

  C-a           prefix

  D             choose-client
  d             detach-client
  C-i           switch-client -p
  C-o           switch-client -n

  s             choose-tree

  c             new-window
  ,             rename-window
  C-hl          select-window
  1..9          select-window
  f             find-window content|...
  C-S-←→        swap-window
  .             command-prompt "move-window -t '%%'"
  |-            split-window
  C-w           kill-window

  hjkl          select-pane
  q             display-panes
  C-A-←↑↓→      resize-pane
  z             resize-pane -Z
  {}            swap-pane
  !             break-pane
  x             kill-pane
  C-A-a         kill-pane -a
  L             pipe-pane
  C-A-s         setw synchronize-panes

  C-c           toggle copy-mode-vi (•)
  • /           search ↓ (vi)
  • ?           search ↑ (vi)
  • v|V|C-v     select
  • ␣           clear selection
  • y           yank selection → tmux buffer, clipit buffer
  • C-j         yank selection → tmux buffer
  • A           yank selection → tmux buffer append
  • Y           yank to EOL → tmux buffer

  =             choose tmux buffer
  p             paste from tmux buffer
  C-v           paste from clipit buffer

  :             command-mode

  r             reload tmux.conf

  C-I           tpm: install
  C-U           tpm: update

  C-s           tmux-resurrect: save
  C-r           tmux-resurrect: restore
