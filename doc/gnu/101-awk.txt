  Uttrykk: $0 ~ /regexp/ ≡ /regexp/

  END[FILE]

  switch(exp) { case konstant: stmt ...; default: stmt }
  for(exp; exp; exp) stmt
  for(key in arr) stmt
    while
    do-while
    break
    continue
    next

    getline
    print exp₁, ... [> "file"]

    delete arr[key]
    { stmt₁; stmt₂; ... }
