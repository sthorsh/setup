Commands
  awk [opts] 'program' [files]
  awk [opts] '[pattern] [{ action }]' [files]

  cut -c list [files]
  cut -d delim -f list [files]

  grep [-EFPr] [opts] regex [files]

  join [opts] file₁ file₂
  
  sed [-En] 'cmd' [files]
  sed [-En] 'cmd ; cmd ...' [files]
  sed [-En] -f script-files [files]

  sort [opts] [files]
  sort -t: /etc/passwd
  sort -t: -k4nr,4 /etc/passwd
  sort -t: -k4nr,4 -u /etc/passwd (uniq keys)
  sort -t: -k4nr,4 --stable /etc/passwd

  tr [opts] char-set₁ char-set₂

  uniq [-c] [-d] [-u] [files] (uniq records)

  wc [opts] [files]
  
---------

awk, cut, join, sort
  Input is stream of records
  Records consists of fields
  Records are delimited by newline (lines)
  Fields are delimited by whitespace or opt (columns)
