ENVIRONMENT

declare
  declare -x                    vis envvars
  declare -x name               eksporter var
  declare -x name=value         eksporter var
  declare +x name               skru av export-attributtet

  declare -fx                   vis envfunksjoner
  declare -fx name              eksporter funksjon
  declare -f +x name            skru av export-attributtet

export
  export                        vis envvars
  export name                   eksporter var
  export name=value             eksporter var
  export -n name                skru av export-attributtet
  
  export -f                     vis envfunksjoner
  export -f name                eksporter funksjon
  export -fn name               skru av export-attributtet

unset
  unset name                    slett envvar
  unset -f name                 slett envfunksjon
