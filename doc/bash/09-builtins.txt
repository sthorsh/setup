BUILTIN CMDs

  alias navn=verdi                  deklarer navn (og definer verdi)

  bg [job_id ...]                   send current-job|job_id ... til bg
  bind -p                           vis keyseq → readline-fn
  bind -v                           vis readline-var → verdi
  bind ks:rf                        bind keyseq → readline-fn
  bind "set rv val"                 bind readline-var → verdi
  break [N]                         kun ifm keywords for, until, while
  builtin b arg ...                 utfør b

  cd arg                            cd til arg
  command c arg ...                 utfør builtin el cmd c
  continue [N]                      kun ifm keywords for, until, while

  declare -p                        vis vardefs
  declare -F                        vis fndecls
  declare ±aAinrx... navn=verdi     dekl|def var NB -n → referanse
  disown job_id|pid                 fornekt jobb

  echo -en arg ...                  skriv til stdout
  enable [-n] b                     enable|disable builtin b
  eval arg                          send arg til prosessering og utførelse
  exec cmd arg ...                  erstatt skall med cmd
  exit [N]                          avslutt skall
  export                            vis eksporterte vars
  export name                       eksporter var
  export navn=verdi                 definer/eksporter var

  false                             returverdi 1
  fg [job_id]                       send current-job|job_id til fg

  getopts optstring name arg        parse opt args

  hash cmd                          legg cmd til hash
  help [b]                          help alle|en builtin
  history                           vis historie

  jobs -l                           vis jobber

  kill [-l] [sig job_id|pid]        vis|send signal(er)

  let arg ...                       eval aritmetiske uttrykk
  local -aAinrx... navn=verdi       dekl|def lokal var NB -n → referanse
  logout [N]                        avslutt login-skall

  mapfile array                     les fra stdin til array

  printf format arg                 skriv til stdout
  pwd                               print working dir

  read name ...                     les fra stdin til name ...
  readonly -aA... navn=verdi        dekl|def read-only var
  return [N]                        avslutt fn|script

  set -o posix; set                 vis vars
  set ±attr | ±o opt                set flag ref echo $-
  set -- arg ...                    set $1,... ref echo $*
  shift [N]                         shift $1,...
  shopt [-p]                        vis shell options
  shopt -s opt                      set shell option
  shopt -u opt                      unset shell option
  source fil arg ...                utfør cmds i fil med $1,...

  test expr                         synonym med [ expr ]
  trap [-l] [cmd sig_id]            vis sig | utfør cmd ved sig_id
  true                              returverdi 0
  type -a name                      alias|builtin|cmd|script

  ulimit [-a] [-b... limit]         vis|endre user resource limits
  umask [-S] [mask]   t             vis|endre user file creation mask
  unalias name                      rm alias
  unset name                        rm var|fn

  wait [-n] [job_id|pid]            vent på alle|neste|job_id|pid
