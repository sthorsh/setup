Instruksjoner
  FROM name:tag               image: base layer

  LABEL name=value            image: config Labels

  ARG name[=value]            build: defined
                              image: n/a
                              run/container: n/a

  ENV key value               build: defined
                              image: config Env
                              run/container: defined

  USER user[:group]           for subsequent commands
                              image: config User
                              run/container: user is required

  WORKDIR path                for subsequent RUN, ADD, COPY, ENTRYPOINT, CMD
                              image: config WorkingDir
                              run/container: directory is created

  RUN cmd && cmd ...          /bin/sh -c cmd && ...
                              image: cmd echo → n/a  
                              image: cmd mkdir → nytt layer

  ADD path|tar|url path       image: nytt layer

  COPY path path              image: nytt layer

  EXPOSE port                 map ports
                              image: config ExposedPorts
                              run -p ...

  VOLUME directory            mount volume
                              image: config Volumes
                              run -v ...

  ENTRYPOINT cmd              exec|shell
                              image: config Entrypoint

  CMD cmd                     exec|param|shell
                              image: config Cmd
