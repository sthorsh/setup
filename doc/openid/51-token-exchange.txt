TOKEN EXCHANGE

Rq /token
  grant_type=token-exchange
  client_id=CLIENT_ID  
  client_secret=CLIENT_SECRET
  scope=SCOPE
  resource=https://rs.domain.tld/api 
  subject_token=SUBJECT_TOKEN
  subject_token_type=id_token|access_token|refresh_token|saml|jwt

OP
  Validerer CLIENT_ID
  Validerer CLIENT_SECRET
  Genererer ACCESS_TOKEN

Rs /token
  access_token: ACCESS_TOKEN
  issued_token_type:access_token
  token_type:Bearer
  expires_in:60
