Opsjoner
  -o set-option         hvis on true

Variabler
  -v var                hvis deklarert return true

Filer
  -e f1                 hvis exists return true
  -f f1                 hvis file return true
  f1 -nt f2             hvis newer-than return true
  f1 -ot f2             hvis older-than return true
      
Aritmetisk
  n₁ -eq n₂             hvis eq return true
  n₁ -ne n₂             hvis ne return true
  n₁ -lt n₂             hvis lt return true
  n₁ -gt n₂             hvis gt return true
  n₁ -le n₂             hvis le return true
  n₁ -ge n₂             hvis ge return true

Params
  -n $s                 hvis def true; false
  -z $s                 hvis udef true; false
  $s₁ = $s₂             hvis lik true; false
  $s₁ != $s₂            hvis ulik true; false
  $s₁ < $s₂             hvis før true; false
  $s₁ < $s₂             hvis etter true; false
  ...
  ${#s}                 lengde
  ...
  ${s:offset:lengde}    delstreng
  ${s₁:-s₂}             hvis def s₁; s₂
  ${s₁:=s₂}             hvis def s₁; tilordne og s₂
  ${s₁:?s₂}             hvis def s₁; exit og s₂
  ${s₁:+s₂}             hvis def s₂; null
  ...
  ${s#p}                del * ? [] fra ^
  ${s##p}               del * ? [] fra ^ (greedy)
  ${s%p}                del * ? [] fra $
  ${s%%p}               del * ? [] fra $ (greedy)
  ${s/p/r}              sub * ? [] med r
  ${s//p/r}             sub * ? [] med r (greedy)

    String extended pattern matching

    Boolske op
      ! exp           true if exp is false
      exp1 -a exp2    exp1 is true and exp2 is true (ikke aritmetikk)
      exp1 -o exp2    exp1 is true or exp2 is true (ikke aritmetikk)

      =   set variabelverdi
      =+  append til variabelverdi
