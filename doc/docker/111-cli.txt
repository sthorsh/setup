  search term                     docker_hub
  login register                  docker.io, ...
  logout register                 https://index.docker.io/v1/, ...
  pull register/repo:tag          get manifest, pull layers, hash content, cache layers

image → container
  create img                      ny container

  run img [cmd]                   search; pull; extract layers; create; start
  run -d ...                      detach
  run -i ...                      interactive
  run -p p₀:p₁ ...                host port p₀ → container port p₁
  run -P ...                      random host ports → exposed container ports
  run -t ...                      tty
  run -v dir₀:dir₁ ...            volume
  run --rm ...                    remove
