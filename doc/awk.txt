Bruk
  awk [opts] 'program' [var=val file ... ]
  awk [opts] -f pf.awk [var=val file ... ]

Opts
  awk --help

Program
  pattern { action }
  pattern { action }
  ...

Operation
  foreach file
    foreach line/record (RS:"\n", FS:" ")
      foreach pattern
        if pattern matches line/record
          do action

  BEGINFILE
  Utrykk
  Pat₁, pat₂
  ENDFILE
  END

Actions
  if(exp) stmt; else stmt
  switch(exp) { case konstant: stmt ...; default: stmt }
  for(exp; exp; exp) stmt
  for(key in assoc_arr) stmt
  while(exp) stmt
  do stmt while(exp)
  break
  continue
  exit
    delete assoc_arr[key]
    getline
    next
    nextfile
    print exp, ... > "file"
    printf "...", exp, ... > "file"
    { stmt; ...}
