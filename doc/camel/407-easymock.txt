EasyMock
  Declare
    @Mock
    @Mock(MockType.NICE)
    @Mock(MockType.STRICT)
  Create
    mock()
    niceMock()
    strictMock()
  Reset
      reset(mock) 
  Record
      anyObject()       arg skal være Object
      anyString()       arg skal være String
      anyDouble()       arg skal være Double
      anyLong()         arg skal være Long
      anyBoolean()      arg skal være Boolean
      contains()        arg skal inneholde...
      eq()              arg skal være lik...
      isNull()          arg skal være null
      notNull()         arg skal ikke være null
      and(m₁, m₂)       arg skal matche m₁ og m₂
      expect(k)         ved kall k,...
      expectLastCall()  ved foregående kall,...
      andReturn(v)      returner v...
      andThrow(e)       kast e...
      times(n)          n ganger
      anyTimes()        vilkårlig antall ganger
  Replay
      replay(mock)
  Test
  Verify
      verify(mock)
