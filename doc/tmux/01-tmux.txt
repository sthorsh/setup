tmux

  ?             list-keys
  C-a           prefix

  D             choose-client
  d             detach-client
  C-i           switch-client -p
  C-o           switch-client -n

  s             choose-tree

  c             new-window
  C-w           kill-window
  ,             rename-window
  C-hl          select-window
  1..9          select-window
  f             find-window content|...
  C-S-←→        swap-window
  .             command-prompt "move-window -t '%%'"
  |-            split-window

  x             kill-pane
  C-A-a         kill-pane -a
  q             display-panes
  hjkl          select-pane
  !             break-pane
  z             resize-pane
  C-A-←↑↓→      resize-pane
  {}            swap-pane
  L             pipe-pane
  C-A-s         setw synchronize-panes

  C-c           copy-mode-vi
  y             selection → tmux buffer, clipit buffer
  C-j           selection → tmux buffer
  A             selection → tmux buffer append
  Y             EOL → tmux buffer

  =             tmux buffer
  p             paste from tmux buffer
  C-v           paste from clipit buffer

  :             command-mode

  r             reload tmux.conf

  C-I           tpm: install
  C-U           tpm: update

  C-s           tmux-resurrect: save
  C-r           tmux-resurrect: restore
